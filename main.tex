% 5
\begin{frame}{Đóng góp}
\small
\begin{enumerate}
  \item \textbf{Multimodal ingestion bảo toàn evidence cho tài liệu đa miền}: PDF, DOCX, PPTX, scan, handwriting, spreadsheet, figure/chart và audio được chuẩn hóa thành evidence unit giàu provenance.
  \item \textbf{Graph-augmented hybrid retrieval lightweight}: BGE-M3 dense+sparse retrieval, RRF fusion, reranking và graph entity/relation/event lưu trong MongoDB.
  \item \textbf{Verification sau generation}: citation injection, SLEC, citation alignment, quality gate và controlled refusal.
  \item \textbf{Table reasoning deterministic tùy chọn}: aggregation/lookup trên full-grid table thay vì để LLM tự tính trên top-k dòng.
\end{enumerate}
\vspace{0.2cm}
\begin{block}{Cách nhấn mạnh điểm mới}
Không tuyên bố thuật toán RAG mới; điểm mới nằm ở \keyword{evidence-preserving system design} cho tài liệu đa miền, đa phương thức.
\end{block}
\end{frame}

% 13
\begin{frame}{Graph-Augmented Retrieval}
\begin{columns}[T,onlytextwidth]
\column{0.32\textwidth}
\begin{block}{Graph-primary}
\small Match entity, duyệt relation 1--2 hop, rồi chuyển path thành graph evidence chunk.
\end{block}
\column{0.32\textwidth}
\begin{block}{Graph boost}
\small Boost chunk sau rerank khi chunk/block trùng với KG relation có confidence cao.
\end{block}
\column{0.32\textwidth}
\begin{block}{Graph probe}
\small Sau context packing ban đầu, thêm một số graph-linked chunk cho câu hỏi nặng về relation.
\end{block}
\end{columns}
\vspace{0.25cm}
\begin{block}{Cách đóng khung chính}
Graph là retrieval-control layer: route, expand, score và probe retrieval nhưng vẫn evidence-grounded.
\end{block}
\end{frame}

  \item Synthesizer trả lời từ packed evidence.
  \item Câu trả lời cuối vẫn đi qua verification.
\end{itemize}
\end{block}
\end{columns}

\begin{block}{Phân biệt quan trọng}
Đây là controlled retrieval planning, không phải autonomous ReAct-style agent.
